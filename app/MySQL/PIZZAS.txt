Table "items_type" {
  "id" int [not null]
  "type" varchar(45) [default: NULL]
 
}



Table "ingredients" {
  "id" int [pk, not null]
  "name" varchar(25) [default: NULL]
  "price" double [not null]
  "veggie" tinyint [not null]
}

Table "items" {
  "id" int [pk, not null, increment]
  "name" varchar(25) [default: NULL]
  "price" double 
}

Table "customers" {
  "id" int [pk, not null]
  "name" varchar(45) [default: NULL]
  "email" varchar(45) [not null]
  "phone" varchar(45) [default: NULL]
  "passwd" varchar(45) [not null]
}



Table "riders" {
  "id" int [pk, not null]
  "name" varchar(45) [default: NULL]
  "available" tinyint [not null, default: "0"]
  "postal_code" int [not null]
}

Table "orders" {
  "id" int [pk, not null]
  "idcustomer" int [pk, not null]
  "idrider" int [not null]
  "price" double [not null]
  "ready_at" timestamp [not null]
  "picked_up_at" timestamp [default: NULL]
  "delivered" tinyint [default: NULL]
  "discount_code" varchar(10) [pk]
}

Ref: items.id < ingredients.id
Ref: items.id - items_type.id

Ref: orders.id <> items.id

Ref: orders.idrider > riders.id
Ref: customers.id <> orders.id
